§ 01 檔案描述符（File Descriptor）

在 Linux 中，所有 I/O 都透過整數索引來操作——這個整數就叫做「檔案描述符」（File Descriptor，簡稱 fd）。不論是磁碟檔案、網路連線、鍵盤、螢幕，核心都統一抽象成同一套介面。

三個預設描述符

每個行程啟動時，核心自動開啟三個 fd：

fd 編號名稱常數（<unistd.h>）預設連接到用途
0stdinSTDIN_FILENO鍵盤讀取輸入
1stdoutSTDOUT_FILENO終端機螢幕輸出結果
2stderrSTDERR_FILENO終端機螢幕輸出錯誤訊息
3+使用者 fd—由 open() 指定任意檔案/pipe
💡 重要觀念： fd 0/1/2 只是整數。若用 dup2() 把它們重新導向到檔案，原本的鍵盤/螢幕連線就被取代，這正是 Shell 重新導向（< >）的底層機制。 
 行程（Process）內部
 ┌──────────────────────────────────────┐
 │  fd 表（File Descriptor Table）        │
 │  ┌───┬──────────────────────────┐    │
 │  │ 0 │ ──────────────────────▶ 鍵盤   │
 │  ├───┼──────────────────────────┤    │
 │  │ 1 │ ──────────────────────▶ 螢幕   │
 │  ├───┼──────────────────────────┤    │
 │  │ 2 │ ──────────────────────▶ 螢幕   │
 │  ├───┼──────────────────────────┤    │
 │  │ 3 │ ──────────────────────▶ test.txt （open後）│
 │  └───┴──────────────────────────┘    │
 └──────────────────────────────────────┘
fd_demo.c — 直接用 write() 寫入 fd 
#include <unistd.h>

int main() {
    /* fd=1 就是 stdout，fd=2 就是 stderr */
    write(1, "這行輸出到 stdout (fd=1)\n", 28);
    write(2, "這行輸出到 stderr (fd=2)\n", 28);
    return 0;
}

§ 02 open / read / write / close

open() — 開啟或建立檔案

函式原型 
#include <fcntl.h>

int open(const char *path, int flags);
int open(const char *path, int flags, mode_t mode);
/* 成功: 回傳 fd (≥3)；失敗: 回傳 -1，並設 errno */
旗標（flags）意義
O_RDONLY唯讀
O_WRONLY唯寫
O_RDWR讀寫
O_CREAT不存在則建立（需提供 mode）
O_TRUNC開啟時清空檔案
O_APPEND每次寫入都追加到尾端
file_rw.c — 完整讀寫範例 
#include <fcntl.h>
#include <unistd.h>
#include <stdio.h>
#include <string.h>

int main() {
    /* ── 1. 寫入檔案 ── */
    int fd = open("hello.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
    if (fd == -1) { perror("open"); return 1; }

    const char *msg = "Hello, Linux File System!\n";
    ssize_t n = write(fd, msg, strlen(msg));
    printf("寫入 %zd 個位元組\n", n);
    close(fd);                   /* ★ 一定要關閉！ */

    /* ── 2. 讀回檔案 ── */
    fd = open("hello.txt", O_RDONLY);
    if (fd == -1) { perror("open"); return 1; }

    char buf[128] = {0};
    n = read(fd, buf, sizeof(buf) - 1);
    printf("讀回：%s", buf);
    close(fd);
    return 0;
}
⚠ 常見錯誤：忘記呼叫 close(fd)。每個 open() 都會佔用一個 fd 槽，不關閉最終會導致「Too many open files」錯誤，且寫入資料可能不被 flush 到磁碟。

read() / write() 回傳值說明

📖

read(fd, buf, count)

從 fd 最多讀 count bytes 到 buf。>0實際讀到的 bytes 0EOF（檔案結束） -1錯誤

✏️

write(fd, buf, count)

從 buf 最多寫 count bytes 到 fd。>0實際寫出的 bytes -1錯誤

🔒

close(fd)

釋放 fd 槽，確保資料 flush。0成功 -1錯誤

§ 03 fork() — 複製行程

fork() 會讓目前的行程（parent）複製出一個幾乎完全相同的子行程（child）。兩者從 fork() 回傳點同時繼續執行，差別只在回傳值。

函式原型 
#include <unistd.h>
#include <sys/types.h>

pid_t fork(void);
/* 在 parent 中回傳 child 的 PID（> 0）
   在 child  中回傳 0
   失敗時回傳 -1（parent 中）                */
PID = 1234
Parent Process
fork() 回傳
child PID = 1235
→ fork() →
PID = 1235
Child Process
fork() 回傳
0

Child 繼承 Parent 的什麼？

📋

繼承（複製）

程式碼、資料（獨立副本）、開啟的 fd 列表、工作目錄、環境變數

🆕

獨立（不同）

PID、記憶體空間（Copy-on-Write）、執行中的 fd 位置游標（某些情況共用）

fork1.c — 基本 fork 用法 
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    pid_t pid = fork();

    if (pid < 0) {
        perror("fork 失敗");
        return 1;

    } else if (pid == 0) {
        /* ── 子行程區塊 ── */
        printf("[Child]  PID=%d, Parent PID=%d\n",
               getpid(), getppid());
        return 0;

    } else {
        /* ── 父行程區塊 ── */
        printf("[Parent] PID=%d, Child PID=%d\n",
               getpid(), pid);
        int status;
        waitpid(pid, &status, 0);   /* 等待子行程結束，回收資源 */
        printf("[Parent] 子行程結束，exit code=%d\n",
               WEXITSTATUS(status));
    }
    return 0;
}
⚙ waitpid() 的重要性：若 parent 不呼叫 waitpid()，已結束的子行程會變成「Zombie（殭屍行程）」，佔用系統 PID 資源直到 parent 結束。

§ 04 execvp() — 執行新程式

exec 系列函式會把目前行程的「程式影像」完全替換成新程式，原本的程式碼不再執行。搭配 fork() 即可在子行程中執行任意命令。

函式原型 
#include <unistd.h>

int execvp(const char *file, char *const argv[]);
/* file  : 程式名稱（自動搜尋 PATH）
   argv  : 參數陣列，argv[0] = 程式名，最後一個必須是 NULL
   成功時不回傳（行程已被取代）；失敗回傳 -1                  */
💡 exec 函式族：execvp（v=vector陣列, p=自動搜PATH）是最常用的。其他變體：execv（需完整路徑）、execl（l=可變參數）、execle（e=自訂環境變數）。
exec1.c — fork + execvp 執行 ls -l 
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    pid_t pid = fork();

    if (pid == 0) {
        /* 子行程：執行 ls -l */
        char *args[] = {"ls", "-l", NULL};  /* ★ 最後必須是 NULL */
        execvp("ls", args);

        /* 若 execvp 回傳到這裡，表示執行失敗 */
        perror("execvp");
        return 1;

    } else if (pid > 0) {
        /* 父行程：等待子行程 */
        waitpid(pid, NULL, 0);
        printf("[Parent] ls 執行完畢\n");
    }
    return 0;
}

fork + exec 的核心流程

Parent 呼叫 fork()，產生 Child（繼承所有 fd）

Child 視需求用 dup2() 重新導向 stdin/stdout

Child 呼叫 execvp()，整個行程影像被新程式取代

Parent 呼叫 waitpid()，等待 Child 結束並回收資源

§ 05 dup2() — 重新導向 fd

dup2(oldfd, newfd) 會讓 newfd 變成 oldfd 的複製品——也就是說，之後對 newfd 的讀寫，實際上會作用在 oldfd 所指向的目標。這是實作 I/O 重新導向的關鍵。

函式原型 
#include <unistd.h>

int dup2(int oldfd, int newfd);
/* 若 newfd 已開啟，先自動 close(newfd) 再複製
   成功: 回傳 newfd；失敗: 回傳 -1              */

重新導向 stdout 到檔案（模擬 ./prog > out.txt）

 執行前：
 fd=1 ──▶ 螢幕
 fd=3 ──▶ out.txt  （open 回傳 3）

 dup2(3, 1) 之後：
 fd=1 ──▶ out.txt  ★ stdout 現在寫入檔案
 fd=3 ──▶ out.txt

 close(3) 之後（清理多餘 fd）：
 fd=1 ──▶ out.txt
 fd=3   已關閉
redirect_out.c — 把子行程輸出導向檔案 
#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    int fd = open("out.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);

    pid_t pid = fork();
    if (pid == 0) {
        /* 子行程：把 stdout(1) 導向 fd（out.txt） */
        dup2(fd, 1);   /* 現在 printf/write(1,...) 都寫到 out.txt */
        close(fd);     /* fd=3 已不需要，關掉 */

        char *args[] = {"ls", "-l", NULL};
        execvp("ls", args);
        perror("execvp");
        return 1;
    }

    close(fd);              /* 父行程也要關閉不需要的 fd */
    waitpid(pid, NULL, 0);
    printf("ls 結果已寫入 out.txt\n");
    return 0;
}
/* Shell 等效：ls -l > out.txt */

重新導向 stdin 從檔案讀取（模擬 ./prog < in.txt）

redirect_in.c — 把 stdin 導向檔案 
#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    int fd = open("input.txt", O_RDONLY);

    pid_t pid = fork();
    if (pid == 0) {
        dup2(fd, 0);    /* stdin(0) 現在從 input.txt 讀 */
        close(fd);

        char *args[] = {"cat", NULL};
        execvp("cat", args);
        perror("execvp");
        return 1;
    }

    close(fd);
    waitpid(pid, NULL, 0);
    return 0;
}
/* Shell 等效：cat < input.txt */

§ 06 pipe() + dup2() — 行程間管線

Shell 中的 cmd1 | cmd2 使用 pipe() 建立一對 fd，再用 dup2() 把 cmd1 的 stdout 接到管線寫端、cmd2 的 stdin 接到管線讀端。

 pipe(pipefd) 建立後：

 pipefd[0] ◀─── PIPE 緩衝區 ◀─── pipefd[1]
 （讀端）                              （寫端）

 dup2 之後：

 cmd1  stdout(1) ─▶ pipefd[1] ─▶ PIPE ─▶ pipefd[0] ─▶ stdin(0) cmd2
pipe1.c — 實作 ls -l | grep .c 
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    int pipefd[2];
    pipe(pipefd);   /* pipefd[0]=讀端, pipefd[1]=寫端 */

    /* ── 子行程1：ls -l（stdout → pipe寫端） ── */
    pid_t pid1 = fork();
    if (pid1 == 0) {
        dup2(pipefd[1], 1);    /* stdout → pipe[1] */
        close(pipefd[0]);      /* 關閉不用的讀端 */
        close(pipefd[1]);
        char *args[] = {"ls", "-l", NULL};
        execvp("ls", args);
        perror("execvp ls"); return 1;
    }

    /* ── 子行程2：grep .c（stdin ← pipe讀端） ── */
    pid_t pid2 = fork();
    if (pid2 == 0) {
        dup2(pipefd[0], 0);    /* stdin ← pipe[0] */
        close(pipefd[1]);      /* 關閉不用的寫端 */
        close(pipefd[0]);
        char *args[] = {"grep", "\\.c", NULL};
        execvp("grep", args);
        perror("execvp grep"); return 1;
    }

    /* ── 父行程：關閉 pipe 並等待 ── */
    close(pipefd[0]);
    close(pipefd[1]);  /* ★ 父行程關閉寫端，grep 才能感知到 EOF */
    waitpid(pid1, NULL, 0);
    waitpid(pid2, NULL, 0);
    return 0;
}
⚠ 管線死結（Deadlock）常見原因：父行程若保留著 pipe 的寫端（pipefd[1]）不關閉，grep（讀端的行程）會永遠等待更多輸入，因為核心認為「還有人持有寫端」，不會發送 EOF → 造成死結。

§ 07 完整綜合範例

範例 A：迷你 Shell（支援 > 輸出重新導向）

minishell.c — 解析並執行帶有 > 的命令 
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/wait.h>

#define MAXARGS 64

void run_cmd(char *argv[], const char *outfile) {
    pid_t pid = fork();
    if (pid == 0) {
        if (outfile) {
            int fd = open(outfile,
                         O_WRONLY | O_CREAT | O_TRUNC, 0644);
            if (fd < 0) { perror("open"); exit(1); }
            dup2(fd, 1);  /* stdout → 檔案 */
            close(fd);
        }
        execvp(argv[0], argv);
        perror("execvp"); exit(1);
    } else {
        waitpid(pid, NULL, 0);
    }
}

int main() {
    char line[512];
    printf("mysh$ ");
    while (fgets(line, sizeof(line), stdin)) {
        line[strcspn(line, "\n")] = '\0';

        /* 分割 > */
        char *outfile = NULL;
        char *redir = strchr(line, '>');
        if (redir) {
            *redir = '\0';
            outfile = redir + 1;
            while (*outfile == ' ') outfile++; /* 去空白 */
        }

        /* Tokenize */
        char *argv[MAXARGS]; int argc = 0;
        char *tok = strtok(line, " \t");
        while (tok && argc < MAXARGS-1) {
            argv[argc++] = tok;
            tok = strtok(NULL, " \t");
        }
        argv[argc] = NULL;
        if (argc == 0) { printf("mysh$ "); continue; }

        run_cmd(argv, outfile);
        printf("mysh$ ");
    }
    return 0;
}

範例 B：使用 write/read 自製 cat 命令

mycat.c — 不用 stdio，純用系統呼叫 
#include <fcntl.h>
#include <unistd.h>

#define BUF 4096

int main(int argc, char *argv[]) {
    int fd = (argc > 1) ? open(argv[1], O_RDONLY) : 0;
    if (fd < 0) { write(2, "open error\n", 11); return 1; }

    char buf[BUF];
    ssize_t n;
    while ((n = read(fd, buf, BUF)) > 0)
        write(1, buf, n);   /* 讀多少寫多少到 stdout */

    if (fd != 0) close(fd);
    return 0;
}

概念總整理

系統呼叫作用重要參數/回傳值常見用途
open() 開啟/建立檔案，取得 fd flags（O_RDONLY…）, mode（0644） 讀寫檔案前的第一步
read() 從 fd 讀資料到 buffer 回傳 0=EOF, -1=錯誤 讀檔案/pipe/stdin
write() 將 buffer 資料寫入 fd 回傳實際寫入 bytes 寫檔案/pipe/stdout
close() 釋放 fd — 每個 open/pipe 後都要呼叫
fork() 複製行程 parent=child PID, child=0 多行程程式的起點
execvp() 取代行程映像執行新程式 argv[] 最後必須是 NULL shell 執行命令
dup2() 複製 fd，實現 I/O 重新導向 dup2(src, dst) > < | 的底層
pipe() 建立行程間通道 [讀端, 寫端] pipefd[0]=讀, pipefd[1]=寫 管線 cmd1 | cmd2
🎯 核心設計模式：Shell 的 I/O 重新導向完全建立在「fd 只是整數，可以被替換」的觀念上。fork()後子行程先用 dup2() 偷偷替換掉 0/1，再呼叫 execvp()——新程式以為自己在讀鍵盤或寫螢幕，實際上卻在讀檔案或寫 pipe。這種「資源繼承 + 重新導向」是 Unix 設計的精髓。

# 8.2 TCP Server/Client 實作

## TCP 連線生命週期

```
SERVER                              CLIENT
  │                                   │
  ├─ socket() ←──────────── socket() ─┤
  ├─ bind()                            │
  ├─ listen()                          │
  ├─ accept() ←──── connect() ────────┤
  │    （阻塞，等待客戶端）    │         │
  │         ←──── 建立連線 ────→       │
  ├─ recv()  ←──── send() ───────────┤
  ├─ send()  ────→ recv() ──────────┤
  ├─ close() ←──── close() ──────────┤
  │                                   │
```

## 完整 Echo Server 實作

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>

#define BUFSIZE 1024

int main(int argc, char *argv[]) {
    if (argc != 2) {
        fprintf(stderr, "用法: %s <port>\n", argv[0]);
        return 1;
    }

    int port = atoi(argv[1]);

    // 1. 建立 Socket
    int server_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (server_fd < 0) { perror("socket"); return 1; }

    // 2. 設定 SO_REUSEADDR（避免 "Address already in use"）
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    // 3. 綁定位址與埠號
    struct sockaddr_in addr = {
        .sin_family = AF_INET,
        .sin_port = htons(port),
        .sin_addr = { htonl(INADDR_ANY) }  // 接受所有介面的連線
    };
    if (bind(server_fd, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
        perror("bind"); return 1;
    }

    // 4. 監聽
    if (listen(server_fd, 5) < 0) { perror("listen"); return 1; }
    printf("Server 啟動，監聽 port %d...\n", port);

    while (1) {
        // 5. 接受連線
        struct sockaddr_in client;
        socklen_t client_len = sizeof(client);
        int client_fd = accept(server_fd,
            (struct sockaddr*)&client, &client_len);

        char client_ip[INET_ADDRSTRLEN];
        inet_ntop(AF_INET, &client.sin_addr, client_ip, sizeof(client_ip));
        printf("新連線: %s:%d\n", client_ip, ntohs(client.sin_port));

        // 6. 資料收發（Echo：收到什麼就送回什麼）
        char buf[BUFSIZE];
        ssize_t n;
        while ((n = recv(client_fd, buf, BUFSIZE, 0)) > 0) {
            send(client_fd, buf, n, 0);
        }

        close(client_fd);
        printf("連線關閉: %s:%d\n", client_ip, ntohs(client.sin_port));
    }

    close(server_fd);
    return 0;
}
```

## 客戶端實作

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>

#define BUFSIZE 1024

int main(int argc, char *argv[]) {
    if (argc != 3) {
        fprintf(stderr, "用法: %s <ip> <port>\n", argv[0]);
        return 1;
    }

    // 1. 建立 Socket
    int fd = socket(AF_INET, SOCK_STREAM, 0);
    if (fd < 0) { perror("socket"); return 1; }

    // 2. 連接到 Server
    struct sockaddr_in addr = {
        .sin_family = AF_INET,
        .sin_port = htons(atoi(argv[2]))
    };
    inet_pton(AF_INET, argv[1], &addr.sin_addr);

    if (connect(fd, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
        perror("connect"); return 1;
    }
    printf("已連接到 %s:%s\n", argv[1], argv[2]);

    // 3. 收發資料
    char buf[BUFSIZE];
    while (1) {
        printf("輸入訊息 (exit 離開): ");
        if (!fgets(buf, BUFSIZE, stdin)) break;

        // 去掉換行
        size_t len = strlen(buf);
        if (len > 0 && buf[len-1] == '\n') buf[len-1] = '\0';

        if (strcmp(buf, "exit") == 0) break;

        send(fd, buf, strlen(buf), 0);
        n = recv(fd, buf, BUFSIZE, 0);
        if (n <= 0) break;
        buf[n] = '\0';
        printf("Server echo: %s\n", buf);
    }

    close(fd);
    return 0;
}
```

## 處理多個客戶端

上述 Server 一次只能服務一個客戶端。要支援多個同時連線，有三種常見模式：

### 1. Fork 模型（每個連線一個程序）

```c
while (1) {
    int client = accept(server_fd, ...);
    if (fork() == 0) {
        close(server_fd);
        // 處理單一客戶端
        handle_client(client);
        close(client);
        exit(0);
    }
    close(client);  // 父程序關閉客戶端 FD
}
```

### 2. 執行緒模型

```c
while (1) {
    int client = accept(server_fd, ...);
    pthread_t t;
    pthread_create(&t, NULL, handle_client, (void*)(intptr_t)client);
    pthread_detach(t);  // 不需要 join
}
```

### 3. I/O 多工（select / poll / epoll）

單一程序同時監聽多個 FD，當資料就緒時處理：

```c
// 簡化示意
fd_set readfds;
FD_ZERO(&readfds);
FD_SET(server_fd, &readfds);

while (1) {
    select(max_fd + 1, &readfds, NULL, NULL, NULL);
    if (FD_ISSET(server_fd, &readfds)) {
        // 有新連線
    }
    // 檢查所有客戶端 FD
}
```

## 思考題

> **Q**：為什麼 `bind()` 前要先設定 `SO_REUSEADDR`？  
> **A**：當 Server 意外終止時，核心可能保留該連線狀態一段時間（TIME_WAIT），導致無法立即重新啟動監聽同一個埠號。`SO_REUSEADDR` 允許核心重新使用該位址，避免「Address already in use」錯誤。
